import matplotlib.pyplot as plt
import numpy as np

# D liu
models = ['Logistic Regression', 'SGDClassifier', 'CNN']
f1_sample = [0.26700788656480173, 0.1706208501822783, 0.3967057347152688]
f1_micro = [0.3340496730420901, 0.21409992921893523, 0.4482074160893031]
f1_macro = [0.1910027212507754, 0.08448656141383039, 0.36230866038772125]

x = np.arange(len(models))
width = 0.3

# To biu 
bars1 = plt.bar(x - width, f1_sample, width, label='F1 Sample', color='skyblue')
bars2 = plt.bar(x, f1_micro, width, label='F1 Micro', color='lightgreen')
bars3 = plt.bar(x + width, f1_macro, width, label='F1 Macro', color='salmon')

for bar in bars1:
    plt.text(bar.get_x() + bar.get_width() / 2, bar.get_height(), f'{bar.get_height():.3f}', 
             ha='center', va='bottom', fontsize=10)

for bar in bars2:
    plt.text(bar.get_x() + bar.get_width() / 2, bar.get_height(), f'{bar.get_height():.3f}', 
             ha='center', va='bottom', fontsize=10)

for bar in bars3:
    plt.text(bar.get_x() + bar.get_width() / 2, bar.get_height(), f'{bar.get_height():.3f}', 
             ha='center', va='bottom', fontsize=10)

plt.xlabel('Gii php')
plt.ylabel('F1 Scores')
plt.title('So snh  o F1 gia cc gii php')
plt.xticks(x, models)
plt.ylim(0, 1) 
plt.legend()

plt.tight_layout()
plt.show()
